CREATE TABLE "business_profiles" (
	"id" uuid PRIMARY KEY DEFAULT gen_random_uuid(),

	"updated_at" timestamp with time zone DEFAULT now() NOT NULL

CREATE TABLE "chat_messages" (
	"id" uuid PRIMARY KEY DEFAULT gen_random_uuid(),
	"session_id" uuid NOT NULL,
	"role" text NOT NULL,
	"message_type" text DEFAULT 'chat_message' NOT NULL,
	"text" text NOT NULL,
	"confidence" numeric(3, 2),
	"suggested_next_questions" text[],
	"created_at" timestamp with time zone DEFAULT now() NOT NULL,
	CONSTRAINT "chat_messages_confidence_check" CHECK (((confidence >= (0)::numeric) AND (confidence <= (1)::numeric))),
	CONSTRAINT "chat_messages_message_type_check" CHECK ((message_type = ANY (ARRAY['chat_message'::text, 'user_message'::text, 'bot_message'::text]))),
	CONSTRAINT "chat_messages_role_check" CHECK ((role = ANY (ARRAY['user'::text, 'bot'::text])))
);
CREATE TABLE "chat_sessions" (
	"id" uuid PRIMARY KEY DEFAULT gen_random_uuid(),
	"business_profile_id" uuid NOT NULL,
	"session_type" text DEFAULT 'demo' NOT NULL,
	"conversation_context" text,
	"device_id" text,
	"started_at" timestamp with time zone DEFAULT now() NOT NULL,
	"expired_at" timestamp with time zone,
	"ended_at" timestamp with time zone,
	"status" text DEFAULT 'active' NOT NULL,
	"feedback_score" integer,
	"feedback_comment" text,
	"created_at" timestamp with time zone DEFAULT now() NOT NULL,
	CONSTRAINT "chat_sessions_feedback_score_check" CHECK (((feedback_score >= 1) AND (feedback_score <= 5))),
	CONSTRAINT "chat_sessions_session_type_check" CHECK ((session_type = ANY (ARRAY['demo'::text, 'interactive'::text, 'full'::text]))),
	CONSTRAINT "chat_sessions_status_check" CHECK ((status = ANY (ARRAY['active'::text, 'ended'::text, 'expired'::text])))
);

);
